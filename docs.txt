API url : host/api

USER:
    register: url: POST 'api/user/register'
        req:{
            name: string, (not null)
            email: string, (not null)
            password: string (min length 6)
        }

        res: {
            success: true,
            id: number,
            email: string,
            name: string,
            token: string (12h)
        }

        error res: {
            success: false,
            message: string,
        }

    login: url: POST 'api/user/login'
        req: {
            email: string, (not null)
            password: string (min length 6)
        }

        res: {
            success: true,
            id: number,
            email: string,
            name: string,
            token: string (12h)
        }

         error res: {
             success: false,
             message: string,
         }

    isAuth: url: GET  'api/user/auth'
        req: {
            headers: {
                Authorization: Bearer token
            }
        }

        res: {
             success: true,
             id: number,
             email: string,
             name: string,
             token: string (12h)
        }

         error res: {
             success: false,
             message: string,
         }
